CREATE TABLE Admin(
	Admin_id CHAR(12)  NOT NULL,
	F_name VARCHAR(50) NOT NULL,
	L_name VARCHAR(50) NOT NULL,
	Email VARCHAR(30) NOT NULL,
	Dob DATE ,

	PRIMARY KEY(Admin_id),

    -- Foreign key 01 : Related to User
  FOREIGN KEY (Admin_id) REFERENCES User(User_id)
        ON UPDATE CASCADE
);
